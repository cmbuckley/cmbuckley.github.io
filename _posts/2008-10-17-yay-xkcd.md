```bash
me:
	@true
a:
	@true
sandwich.:
	@[ -w /etc/shadow ] && echo "Okay." || echo "What? Make it yourself."
```

```bash
$ make me a sandwich.
What? Make it yourself.

$ sudo make me a sandwich.
Okay.
```
